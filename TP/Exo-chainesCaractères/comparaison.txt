#include <stdio.h>
#include <stdlib.h>


int tris(char *str1, char *str2)
{
    int i;
	
	i = 0;
    while((str1[i] == str2[i]) && (str1[i] !='\0'))
		{
		i++;
		}
    if(str1[i] > str2[i])
		{
			return(0);
        }
    else if(str1[i] < str2[i])
		{
            return(1);
        }

}
int main(void)
{
    char str1[20];
    char str2[20];
    int res;

	res = 0;
    printf("Entrer le nom1\n");
    scanf("%s", &str1);
    printf("Entrer le nom2\n");
    scanf("%s", &str2);
    res = tris(*str1,*str2);
    if (res == 0)
		{
			printf("Les noms on était rentré dans le mauvais ordre\n");
		}
     if (res == 1)
		{
			printf("Les noms on était rentré dans le bon ordre\n");
		}
	return 0;
}
