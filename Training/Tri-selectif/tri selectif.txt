ALGORITHME : tri sélectif

VAR

T[0..200]:entier
i,j:entier   //variables de parcours de tableau
n:entier     //censé être égal à 200
indice:entier    //contient l'indice de la case contenant la valeur la plus petite
tmp:entier       //temporaire

DEBUT

pour i=1 à n faire
	//recherche de la plus petite valeur
	indice <-- (i-1)
	pour j=1 à n-(i-1) faire
		si T[indice] > T[i-1 + j]
		alors indice <-- i - 1 + j
		finsi
	finpour

	//échange de valeur
	tmp <-- T[i-1]
	T[i-1] <-- T[indice]
	T[indice] <-- tmp

finpour

FIN
